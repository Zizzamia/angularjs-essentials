Isolate scope

angular.module('tasteAngulajs.filters', [])
  .directive('tweet', function(){
    return {
      restrict: 'A',
      scope: {
        'data': '=',
        'removeTweet': '&remove',
        'removeButtonColor': '@'
      },
      template: '<div class="tweet">'+
      '  <div class="tweet-content">'+
      '    {{ data.text }}'+
      '    <button type="button" class="btn btn-default btn-xs">'+
      '      <span class="fa fa-remove {{ removeButtonColor }}" '+
      '      ng-click="removeTweet($index)"></span>'+
      '    </button>'+
      '  </div>'+
      '</div>'
    };
  })

Directive that wraps arbitrary content

controller

angular.module('tasteAngulajs.filters', [])
  .directive('timeline', function(timelineAPI){
    return {
      restrict: 'A',
      scope: true,
      controller: function($scope) {
        $scope.listTweets = [];
        this.getTimeline = function () {
          $scope.listTweets = timelineAPI.get();
        };
        this.getTimeline(); 
      }
    };
  })

Providers

provider

angular.module('tasteAngulajs.filters', [])
  .provider('timelineAPI', function () {
    var username = '';
    return {
      setUsername: function (value) {
        username = value
      },
      $get: function () {
        var listTweets = [
          { 'username': username, 'text': 'Ciao', 'timestamp': 1410474730056 },
          { 'username': username, 'text': 'Ciao due', 'timestamp': 1410474617870 }
        ];
        return {
          'get': function () {
            return listTweets;
          }, ...
        };
      }
    }
  })
